528      private void getLocationPermission() { 
529      /* 
530       * Request location permission, so that we can get the location of the 
531       * device. The result of the permission request is handled by a callback, 
532       * onRequestPermissionsResult. 
533       */ 
534          if (ContextCompat.checkSelfPermission(this.getApplicationContext(), 
535                  android.Manifest.permission.ACCESS_FINE_LOCATION) 
536                  == PackageManager.PERMISSION_GRANTED) { 
537              mLocationPermissionGranted = true; 
538          } else { 
539              ActivityCompat.requestPermissions(this, 
540                      new String[]{android.Manifest.permission.ACCESS_FINE_LOCATION}, 
541                      PERMISSIONS_REQUEST_ACCESS_FINE_LOCATION); 
542          } 
543      }

546      @Override 
547      public void onRequestPermissionsResult(int requestCode, 
548                                             @NonNull String permissions[], 
549                                             @NonNull int[] grantResults) { 
550          mLocationPermissionGranted = false; 
551          switch (requestCode) { 
552              case PERMISSIONS_REQUEST_ACCESS_FINE_LOCATION: { 
553                  // If request is cancelled, the result arrays are empty. 
554                  if (grantResults.length > 0 
555                          && grantResults[0] == PackageManager.PERMISSION_GRANTED) { 
556                      mLocationPermissionGranted = true; 
557                  } 
558              } 
559          } 
560          updateLocationUI(); 
561      }

564      private void updateLocationUI() { 
565          if (mMap == null) { 
566              return; 
567          } 
568          try { 
569              if (mLocationPermissionGranted) { 
570                  mMap.setMyLocationEnabled(true); 
571                  mMap.getUiSettings().setMyLocationButtonEnabled(true); 
572              } else { 
573                  mMap.setMyLocationEnabled(false); 
574                  mMap.getUiSettings().setMyLocationButtonEnabled(false); 
575                 // mLastKnownLocation = null; 
576                  getLocationPermission(); 
577              } 
578          } catch (SecurityException e)  { 
579              Log.e("Exception: %s", e.getMessage()); 
580          } 
581      }
